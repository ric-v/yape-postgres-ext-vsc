SELECT
  c.id, c.name,
  SUM(o.amount) AS total
FROM
  @public.customers c
JOIN orders o ON c.id = o.customer_id
WHERE o.created_at >= NOW() - INTERVAL '30 days'
GROUP BY c.id, c.name
ORDER BY total DESC LIMIT 10;
